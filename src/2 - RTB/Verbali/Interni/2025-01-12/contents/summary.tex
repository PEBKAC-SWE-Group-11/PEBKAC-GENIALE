    \begin{itemize}
        \item Completata integrazione del chunking semantico;
        \item Completata scrittura UI in Angular;
        \item Iniziata scrittura delle API;
    \end{itemize}
    \item Viene implementata la github action di conversione di file LaTeX a Pdf

\end{itemize}
